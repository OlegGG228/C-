bool isPalindrome(string str) {
    int len = str.length();
    for (int i = 0; i < len / 2; i++) {
        if (str[i] != str[len - 1 - i]) {
            return false;
        }
    }
    return true;
}

#include <iostream>
#include <string>
#include <vector>

using namespace std;

struct Product {
    string name;
    float price;
    int expirationDate;
};

int main() {
    vector<Product> products;

    // запрос количества товаров у пользователя
    int n;
    cout << "Введите количество товаров: ";
    cin >> n;

    // добавление товаров в список
    for (int i = 0; i < n; i++) {
        Product p;
        cout << "Введите наименование товара: ";
        cin >> p.name;
        cout << "Введите стоимость товара: ";
        cin >> p.price;
        cout << "Введите срок годности товара (в днях): ";
        cin >> p.expirationDate;
        products.push_back(p);
    }

    // запрос наименования товара у пользователя
    string inputName;
    cout << "Введите наименование товара: ";
    cin >> inputName;

    // вывод информации о товарах с указанным наименованием
    bool found = false;
    for (int i = 0; i < products.size(); i++) {
        if (products[i].name == inputName) {
            cout << "Наименование товара: " << products[i].name << endl;
            cout << "Стоимость товара: " << products[i].price << endl;
            cout << "Срок годности товара: " << products[i].expirationDate << " дней" << endl;
            cout << endl;
            found = true;
        }
    }

    // сообщение о том, что товар не найден
    if (found == false) {
        cout << "Товар с указанным наименованием не найден." << endl;
    }

    return 0;
}

class String {
private:
    char* str;
    int len;
public:
    String(const char* s) {
        len = strlen(s);
        str = new char[len + 1];
        strcpy(str, s);
    }

    ~String() {
        delete[] str;
        str = NULL;
    }

    int length() const {
        return len;
    }

    void print() const {
        std::cout << str << std::endl;
    }

    char& operator[](int index) {
        return str[index];
    }

    const char& operator[](int index) const {
        return str[index];
    }

    void reverse() {
        for (int i = 0; i < len / 2; i++) {
            char temp = str[i];
            str[i] = str[len - i - 1];
            str[len - i - 1] = temp;
        }
    }

    String substr(int start, int length) const {
        char* temp = new char[length + 1];
        strncpy(temp, str + start, length);
        temp[length] = '\0';
        String result(temp);
        delete[] temp;
        return result;
    }

    bool contains(const char* s) const {
        return strstr(str, s) != nullptr;
    }
};
